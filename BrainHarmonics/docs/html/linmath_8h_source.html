555 LINMATH_H_FUNC void quat_from_mat4x4(quat q, mat4x4 M)
556 {
557  float r=0.f;
558  int i;
559 
560  int perm[] = { 0, 1, 2, 0, 1 };
561  int *p = perm;
562 
563  for(i = 0; i<3; i++) {
564  float m = M[i][i];
565  if( m < r )
566  continue;
567  m = r;
568  p = &perm[i];
569  }
570 
571  r = sqrtf(1.f + M[p[0]][p[0]] - M[p[1]][p[1]] - M[p[2]][p[2]] );
572 
573  if(r < 1e-6) {
574  q[0] = 1.f;
575  q[1] = q[2] = q[3] = 0.f;
576  return;
577  }
578 
579  q[0] = r/2.f;
580  q[1] = (M[p[0]][p[1]] - M[p[1]][p[0]])/(2.f*r);
581  q[2] = (M[p[2]][p[0]] - M[p[0]][p[2]])/(2.f*r);
582  q[3] = (M[p[2]][p[1]] - M[p[1]][p[2]])/(2.f*r);
583 }
